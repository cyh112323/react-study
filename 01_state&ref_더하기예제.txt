■ state 변수
  * 값이 변하면 화면이 변한다
    -함수를 임포트 해야함
    import React, { useState } from 'react';
    - 선언
    let [_name, _setName] = useState('홍길동')  //let name = '홍길동'
    - 변경
    _setName(e.target.innerText)
